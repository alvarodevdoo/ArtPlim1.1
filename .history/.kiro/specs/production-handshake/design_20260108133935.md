```typescript
export class WebSocketServer {
  private io: Server;
  private organizationRooms: Map<string, Set<string>>;
  
  constructor(httpServer: http.Server) {
    this.io = new Server(httpServer, {
      cors: { origin: process.env.FRONTEND_URL }
    });
    this.organizationRooms = new Map();
    this.setupEventHandlers();
  }
  
  private setupEventHandlers() {
    this.io.on('connection', (socket) => {
      socket.on('join-organization', (orgId) => {
        this.joinOrganizationRoom(socket, orgId);
      });
      
      socket.on('disconnect', () => {
        this.handleDisconnect(socket);
      });
    });
  }
  
  public notifyOrganization(orgId: string, event: string, data: any) {
    this.io.to(`org-${orgId}`).emit(event, data);
  }
}
```

```typescript
// NotificationService.ts
export class NotificationService {
  constructor(
    private websocketServer: WebSocketServer,
    private notificationRepository: NotificationRepository
  ) {}
  
  async notifyChangeRequest(pendingChange: PendingChange) {
    const notification = await this.notificationRepository.create({
      organizationId: pendingChange.organizationId,
      type: 'CHANGE_REQUEST',
      title: `Alteração solicitada - Pedido #${pendingChange.order.number}`,
      message: `${pendingChange.requestedBy.name} solicitou alteração`,
      data: { pendingChangeId: pendingChange.id }
    });
    
    this.websocketServer.notifyOrganization(
      pendingChange.organizationId,
      'change-request',
      notification
    );
  }
  
  async notifyChangeDecision(pendingChange: PendingChange, approved: boolean) {
    const notification = await this.notificationRepository.create({
      organizationId: pendingChange.organizationId,
      userId: pendingChange.requestedBy.id,
      type: approved ? 'CHANGE_APPROVED' : 'CHANGE_REJECTED',
      title: `Alteração ${approved ? 'aprovada' : 'rejeitada'}`,
      message: `Sua solicitação foi ${approved ? 'aprovada' : 'rejeitada'}`,
      data: { pendingChangeId: pendingChange.id }
    });
    
    this.websocketServer.notifyOrganization(
      pendingChange.organizationId,
      'change-decision',
      notification
    );
  }
}
```

### 3. Pending Changes Service

**Localização**: `backend/src/modules/production/services/`

```typescript
// PendingChangesService.ts
export class PendingChangesService {
  constructor(
    private pendingChangesRepository: PendingChangesRepository,
    private orderRepository: OrderRepository,
    private notificationService: NotificationService
  ) {}
  
  async createPendingChange(
    orderId: string,
    changes: any,
    requestedBy: string,
    organizationId: string
  ): Promise<PendingChange> {
    const order = await this.orderRepository.findById(orderId);
    if (!order) throw new Error('Pedido não encontrado');
    
    if (order.status !== 'IN_PRODUCTION') {
      throw new Error('Apenas pedidos em produção podem ter alterações pendentes');
    }
    
    const pendingChange = await this.pendingChangesRepository.create({
      orderId,
      organizationId,
      requestedBy,
      changes,
      originalData: order.toJSON(),
      status: 'PENDING',
      priority: this.calculatePriority(changes)
    });
    
    await this.notificationService.notifyChangeRequest(pendingChange);
    
    return pendingChange;
  }
  
  async approveChange(
    pendingChangeId: string,
    reviewedBy: string,
    comments?: string
  ): Promise<void> {
    const pendingChange = await this.pendingChangesRepository.findById(pendingChangeId);
    if (!pendingChange) throw new Error('Alteração não encontrada');
    
    // Aplicar alterações ao pedido
    await this.applyChangesToOrder(pendingChange);
    
    // Atualizar status da alteração
    await this.pendingChangesRepository.update(pendingChangeId, {
      status: 'APPROVED',
      reviewedBy,
      reviewedAt: new Date(),
      reviewComments: comments
    });
    
    await this.notificationService.notifyChangeDecision(pendingChange, true);
  }
  
  async rejectChange(
    pendingChangeId: string,
    reviewedBy: string,
    comments: string
  ): Promise<void> {
    const pendingChange = await this.pendingChangesRepository.findById(pendingChangeId);
    if (!pendingChange) throw new Error('Alteração não encontrada');
    
    await this.pendingChangesRepository.update(pendingChangeId, {
      status: 'REJECTED',
      reviewedBy,
      reviewedAt: new Date(),
      reviewComments: comments
    });
    
    await this.notificationService.notifyChangeDecision(pendingChange, false);
  }
  
  private async applyChangesToOrder(pendingChange: PendingChange): Promise<void> {
    const { orderId, changes } = pendingChange;
    
    // Aplicar cada alteração de forma segura
    for (const [field, newValue] of Object.entries(changes)) {
      await this.orderRepository.updateField(orderId, field, newValue);
    }
    
    // Registrar no histórico do pedido
    await this.orderRepository.addHistoryEntry(orderId, {
      action: 'CHANGE_APPLIED',
      description: 'Alteração aprovada pela produção aplicada',
      userId: pendingChange.reviewedBy,
      data: { pendingChangeId: pendingChange.id }
    });
  }
}
```

```typescript
export const productionRoutes = Router();

// Listar alterações pendentes
productionRoutes.get('/pending-changes', async (req, res) => {
  const { organizationId } = req.user;
  const { status, priority } = req.query;
  
  const pendingChanges = await pendingChangesService.findByOrganization(
    organizationId,
    { status, priority }
  );
  
  res.json(pendingChanges);
});

// Aprovar alteração
productionRoutes.post('/pending-changes/:id/approve', async (req, res) => {
  const { id } = req.params;
  const { comments } = req.body;
  const { userId } = req.user;
  
  await pendingChangesService.approveChange(id, userId, comments);
  
  res.json({ success: true });
});

// Rejeitar alteração
productionRoutes.post('/pending-changes/:id/reject', async (req, res) => {
  const { id } = req.params;
  const { comments } = req.body;
  const { userId } = req.user;
  
  if (!comments) {
    return res.status(400).json({ error: 'Comentário é obrigatório para rejeição' });
  }
  
  await pendingChangesService.rejectChange(id, userId, comments);
  
  res.json({ success: true });
});

// Listar notificações
productionRoutes.get('/notifications', async (req, res) => {
  const { organizationId, userId } = req.user;
  const { unreadOnly } = req.query;
  
  const notifications = await notificationService.findByUser(
    organizationId,
    userId,
    { unreadOnly: unreadOnly === 'true' }
  );
  
  res.json(notifications);
});

// Marcar notificação como lida
productionRoutes.post('/notifications/:id/read', async (req, res) => {
  const { id } = req.params;
  
  await notificationService.markAsRead(id);
  
  res.json({ success: true });
});
```

```typescript
// useWebSocket.ts
export const useWebSocket = () => {
  const [socket, setSocket] = useState<Socket | null>(null);
  const [connected, setConnected] = useState(false);
  const { user } = useAuth();
  
  useEffect(() => {
    if (!user?.organizationId) return;
    
    const newSocket = io(process.env.REACT_APP_WS_URL || 'http://localhost:3001');
    
    newSocket.on('connect', () => {
      setConnected(true);
      newSocket.emit('join-organization', user.organizationId);
    });
    
    newSocket.on('disconnect', () => {
      setConnected(false);
    });
    
    setSocket(newSocket);
    
    return () => {
      newSocket.close();
    };
  }, [user?.organizationId]);
  
  const subscribe = useCallback((event: string, handler: (data: any) => void) => {
    if (socket) {
      socket.on(event, handler);
      return () => socket.off(event, handler);
    }
  }, [socket]);
  
  return { socket, connected, subscribe };
};
```

```typescript
// NotificationContext.tsx
export const NotificationProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [notifications, setNotifications] = useState<Notification[]>([]);
  const { subscribe } = useWebSocket();
  const { toast } = useToast();
  
  useEffect(() => {
    const unsubscribe = subscribe?.('change-request', (notification: Notification) => {
      setNotifications(prev => [notification, ...prev]);
      
      // Notificação visual
      toast({
        title: notification.title,
        description: notification.message,
        variant: 'info'
      });
      
      // Notificação sonora (opcional)
      if ('Notification' in window && Notification.permission === 'granted') {
        new Notification(notification.title, {
          body: notification.message,
          icon: '/favicon.ico'
        });
      }
    });
    
    return unsubscribe;
  }, [subscribe, toast]);
  
  const markAsRead = async (notificationId: string) => {
    await api.post(`/production/notifications/${notificationId}/read`);
    setNotifications(prev => 
      prev.map(n => n.id === notificationId ? { ...n, read: true } : n)
    );
  };
  
  return (
    <NotificationContext.Provider value={{ notifications, markAsRead }}>
      {children}
    </NotificationContext.Provider>
  );
};
```

```typescript
// Producao.tsx
export const Producao: React.FC = () => {
  const [pendingChanges, setPendingChanges] = useState<PendingChange[]>([]);
  const [selectedChange, setSelectedChange] = useState<PendingChange | null>(null);
  const [loading, setLoading] = useState(true);
  const { subscribe } = useWebSocket();
  
  useEffect(() => {
    loadPendingChanges();
    
    const unsubscribe = subscribe?.('change-request', () => {
      loadPendingChanges(); // Recarregar lista
    });
    
    return unsubscribe;
  }, [subscribe]);
  
  const loadPendingChanges = async () => {
    try {
      const response = await api.get('/production/pending-changes');
      setPendingChanges(response.data);
    } catch (error) {
      console.error('Erro ao carregar alterações pendentes:', error);
    } finally {
      setLoading(false);
    }
  };
  
  const handleApprove = async (changeId: string, comments?: string) => {
    try {
      await api.post(`/production/pending-changes/${changeId}/approve`, { comments });
      await loadPendingChanges();
      setSelectedChange(null);
      toast({ title: 'Alteração aprovada com sucesso' });
    } catch (error) {
      toast({ title: 'Erro ao aprovar alteração', variant: 'destructive' });
    }
  };
  
  const handleReject = async (changeId: string, comments: string) => {
    try {
      await api.post(`/production/pending-changes/${changeId}/reject`, { comments });
      await loadPendingChanges();
      setSelectedChange(null);
      toast({ title: 'Alteração rejeitada' });
    } catch (error) {
      toast({ title: 'Erro ao rejeitar alteração', variant: 'destructive' });
    }
  };
  
  return (
    <div className="container mx-auto p-6">
      <div className="flex justify-between items-center mb-6">
        <h1 className="text-2xl font-bold">Painel de Produção</h1>
        <Badge variant="secondary">
          {pendingChanges.length} alterações pendentes
        </Badge>
      </div>
      
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
        {/* Lista de alterações pendentes */}
        <Card>
          <CardHeader>
            <CardTitle>Alterações Pendentes</CardTitle>
          </CardHeader>
          <CardContent>
            {loading ? (
              <div className="flex justify-center p-4">
                <Loader2 className="h-6 w-6 animate-spin" />
              </div>
            ) : pendingChanges.length === 0 ? (
              <p className="text-muted-foreground text-center p-4">
                Nenhuma alteração pendente
              </p>
            ) : (
              <div className="space-y-3">
                {pendingChanges.map(change => (
                  <PendingChangeCard
                    key={change.id}
                    change={change}
                    onSelect={setSelectedChange}
                    selected={selectedChange?.id === change.id}
                  />
                ))}
              </div>
            )}
          </CardContent>
        </Card>
        
        {/* Detalhes da alteração selecionada */}
        <Card>
          <CardHeader>
            <CardTitle>Detalhes da Alteração</CardTitle>
          </CardHeader>
          <CardContent>
            {selectedChange ? (
              <ChangeDetailsPanel
                change={selectedChange}
                onApprove={handleApprove}
                onReject={handleReject}
              />
            ) : (
              <p className="text-muted-foreground text-center p-4">
                Selecione uma alteração para ver os detalhes
              </p>
            )}
          </CardContent>
        </Card>
      </div>
    </div>
  );
};
```

```typescript
// Adicionar ao UpdateOrderUseCase.ts
export class UpdateOrderUseCase {
  constructor(
    private orderRepository: OrderRepository,
    private pendingChangesService: PendingChangesService
  ) {}
  
  async execute(orderId: string, updates: any, userId: string): Promise<Order> {
    const order = await this.orderRepository.findById(orderId);
    if (!order) throw new Error('Pedido não encontrado');
    
    // Se pedido está em produção, criar alteração pendente
    if (order.status === 'IN_PRODUCTION') {
      await this.pendingChangesService.createPendingChange(
        orderId,
        updates,
        userId,
        order.organizationId
      );
      
      // Retornar pedido sem alterações, mas com indicador de pendência
      return { ...order, hasPendingChanges: true };
    }
    
    // Para outros status, aplicar alterações normalmente
    return await this.orderRepository.update(orderId, updates);
  }
}
```

## Fluxo de Dados

### 1. Solicitação de Alteração
```
Vendedor → Frontend → API → PendingChangesService → Database
                                     ↓
                            NotificationService → WebSocket → Operadores
```

### 2. Aprovação/Rejeição
```
Operador → Frontend → API → PendingChangesService → Database
                                     ↓
                            OrderRepository (se aprovado)
                                     ↓
                            NotificationService → WebSocket → Solicitante
```

## Considerações de Performance

### 1. WebSocket
- Usar rooms para isolamento por organização
- Implementar heartbeat para detectar conexões perdidas
- Limitar número de conexões simultâneas por usuário

### 2. Database
- Índices em `organizationId`, `status`, `createdAt`
- Limpeza periódica de alterações antigas
- Paginação para listas grandes

### 3. Frontend
- Debounce em atualizações de lista
- Lazy loading para detalhes de alterações
- Cache local de notificações

## Segurança

### 1. Autenticação WebSocket
- Validar token JWT na conexão
- Verificar permissões antes de entrar em rooms

### 2. Autorização
- Apenas OPERATOR e ADMIN podem aprovar
- Validar organizationId em todas as operações
- Logs de auditoria completos

### 3. Validação de Dados
- Sanitizar todas as entradas
- Validar estrutura de alterações
- Prevenir ataques de injeção
