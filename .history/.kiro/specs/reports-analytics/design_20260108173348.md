```typescript
// AnalyticsEngine.ts
export class AnalyticsEngine {
  constructor(
    private prisma: PrismaClient,
    private cacheService: CacheService,
    private reportRepository: ReportRepository
  ) {}
  
  async generateDashboardData(
    organizationId: string,
    filters: DashboardFilters
  ): Promise<DashboardData> {
    const cacheKey = this.generateCacheKey('dashboard', organizationId, filters);
    
    // Tentar buscar do cache primeiro
    const cached = await this.cacheService.get(cacheKey);
    if (cached) return cached;
    
    // Gerar dados se não estiver em cache
    const data = await this.computeDashboardMetrics(organizationId, filters);
    
    // Cachear por 5 minutos
    await this.cacheService.set(cacheKey, data, 300);
    
    return data;
  }
  
  private async computeDashboardMetrics(
    organizationId: string,
    filters: DashboardFilters
  ): Promise<DashboardData> {
    const { startDate, endDate } = filters;
    
    // Executar queries em paralelo para performance
    const [
      salesMetrics,
      costAnalysis,
      productionMetrics,
      materialMetrics
    ] = await Promise.all([
      this.getSalesMetrics(organizationId, startDate, endDate),
      this.getCostAnalysis(organizationId, startDate, endDate),
      this.getProductionMetrics(organizationId, startDate, endDate),
      this.getMaterialMetrics(organizationId, startDate, endDate)
    ]);
    
    return {
      sales: salesMetrics,
      costs: costAnalysis,
      production: productionMetrics,
      materials: materialMetrics,
      generatedAt: new Date()
    };
  }
}
```

### 3. Cache Service

```typescript
// CacheService.ts
export class CacheService {
  private redis: Redis;
  
  constructor() {
    this.redis = new Redis({
      host: process.env.REDIS_HOST || 'localhost',
      port: parseInt(process.env.REDIS_PORT || '6379'),
      retryDelayOnFailover: 100,
      maxRetriesPerRequest: 3
    });
  }
  
  async get<T>(key: string): Promise<T | null> {
    try {
      const cached = await this.redis.get(key);
      return cached ? JSON.parse(cached) : null;
    } catch (error) {
      console.error('Cache get error:', error);
      return null;
    }
  }
  
  async set(key: string, value: any, ttlSeconds: number = 3600): Promise<void> {
    try {
      await this.redis.setex(key, ttlSeconds, JSON.stringify(value));
    } catch (error) {
      console.error('Cache set error:', error);
    }
  }
  
  async invalidatePattern(pattern: string): Promise<void> {
    try {
      const keys = await this.redis.keys(pattern);
      if (keys.length > 0) {
        await this.redis.del(...keys);
      }
    } catch (error) {
      console.error('Cache invalidation error:', error);
    }
  }
}
```

```typescript
// BackgroundJobService.ts
export class BackgroundJobService {
  private queue: Queue;
  
  constructor() {
    this.queue = new Queue('analytics', {
      redis: {
        host: process.env.REDIS_HOST || 'localhost',
        port: parseInt(process.env.REDIS_PORT || '6379')
      }
    });
    
    this.setupProcessors();
  }
  
  private setupProcessors() {
    // Processador para refresh de views materializadas
    this.queue.process('refresh-materialized-views', async (job) => {
      const { organizationId } = job.data;
      await this.refreshMaterializedViews(organizationId);
    });
    
    // Processador para geração de relatórios
    this.queue.process('generate-report', async (job) => {
      const { reportId, type, organizationId, filters } = job.data;
      await this.generateAndStoreReport(reportId, type, organizationId, filters);
    });
    
    // Processador para export de dados
    this.queue.process('export-data', async (job) => {
      const { exportId, data, format } = job.data;
      await this.exportToFile(exportId, data, format);
    });
  }
  
  async scheduleReportGeneration(
    reportId: string,
    type: ReportType,
    organizationId: string,
    filters: any
  ): Promise<void> {
    await this.queue.add('generate-report', {
      reportId,
      type,
      organizationId,
      filters
    }, {
      attempts: 3,
      backoff: 'exponential',
      delay: 1000
    });
  }
}
```

## Otimizações de Performance

### 1. Views Materializadas

```sql
-- View para métricas de vendas (atualizada a cada hora)
CREATE MATERIALIZED VIEW SalesMetrics AS
SELECT 
  o.organizationId,
  DATE_TRUNC('day', o.createdAt) as date,
  COUNT(*) as order_count,
  SUM(o.total) as total_revenue,
  AVG(o.total) as avg_order_value,
  COUNT(CASE WHEN o.status = 'DELIVERED' THEN 1 END) as delivered_count,
  SUM(CASE WHEN o.status = 'DELIVERED' THEN o.total ELSE 0 END) as delivered_revenue
FROM "Order" o
WHERE o.status != 'CANCELLED'
GROUP BY o.organizationId, DATE_TRUNC('day', o.createdAt);

-- Índice para performance
CREATE INDEX idx_sales_metrics_org_date ON SalesMetrics(organizationId, date);

-- View para análise de custos (atualizada diariamente)
CREATE MATERIALIZED VIEW CostAnalysis AS
SELECT 
  o.organizationId,
  oi.productId,
  p.name as product_name,
  DATE_TRUNC('month', o.createdAt) as month,
  COUNT(oi.id) as item_count,
  SUM(oi.quantity) as total_quantity,
  SUM(oi.costPrice * oi.quantity) as total_cost,
  SUM(oi.unitPrice * oi.quantity) as total_revenue,
  SUM((oi.unitPrice - oi.costPrice) * oi.quantity) as total_margin,
  AVG((oi.unitPrice - oi.costPrice) / oi.unitPrice * 100) as avg_margin_percentage
FROM "Order" o
JOIN "OrderItem" oi ON o.id = oi.orderId
JOIN "Product" p ON oi.productId = p.id
WHERE o.status = 'DELIVERED'
GROUP BY o.organizationId, oi.productId, p.name, DATE_TRUNC('month', o.createdAt);

-- View para análise de materiais
CREATE MATERIALIZED VIEW MaterialAnalysis AS
SELECT 
  pc.organizationId,
  pc.materialId,
  m.name as material_name,
  DATE_TRUNC('month', o.createdAt) as month,
  SUM(
    CASE 
      WHEN m.format = 'SHEET' THEN (oi.width * oi.height / 1000000) * oi.quantity
      WHEN m.format = 'ROLL' THEN (oi.width / 1000) * oi.quantity
      ELSE oi.quantity
    END
  ) as theoretical_consumption,
  SUM(
    CASE 
      WHEN m.format = 'SHEET' THEN (oi.width * oi.height / 1000000) * oi.quantity * (1 + pc.wastePercentage / 100)
      WHEN m.format = 'ROLL' THEN (oi.width / 1000) * oi.quantity * (1 + pc.wastePercentage / 100)
      ELSE oi.quantity * (1 + pc.wastePercentage / 100)
    END
  ) as estimated_consumption,
  AVG(pc.wastePercentage) as avg_waste_percentage
FROM "Order" o
JOIN "OrderItem" oi ON o.id = oi.orderId
JOIN "ProductComponent" pc ON oi.productId = pc.productId
JOIN "Material" m ON pc.materialId = m.id
WHERE o.status IN ('IN_PRODUCTION', 'FINISHED', 'DELIVERED')
GROUP BY pc.organizationId, pc.materialId, m.name, DATE_TRUNC('month', o.createdAt);
```

```typescript
// CacheStrategy.ts
export class CacheStrategy {
  // Cache de curta duração para dados em tempo real
  static readonly REALTIME_TTL = 60; // 1 minuto
  
  // Cache de média duração para dashboards
  static readonly DASHBOARD_TTL = 300; // 5 minutos
  
  // Cache de longa duração para relatórios históricos
  static readonly REPORT_TTL = 3600; // 1 hora
  
  static generateKey(type: string, organizationId: string, params: any): string {
    const paramString = JSON.stringify(params, Object.keys(params).sort());
    return `${type}:${organizationId}:${createHash('md5').update(paramString).digest('hex')}`;
  }
  
  static shouldCache(dataSize: number, computationTime: number): boolean {
    // Cachear se os dados são grandes ou o processamento é lento
    return dataSize > 1000 || computationTime > 1000; // 1MB ou 1 segundo
  }
}
```

```typescript
// OptimizedQueries.ts
export class OptimizedQueries {
  
  // Query otimizada para dashboard com agregações
  static async getDashboardMetrics(
    prisma: PrismaClient,
    organizationId: string,
    startDate: Date,
    endDate: Date
  ) {
    // Usar view materializada quando possível
    const salesMetrics = await prisma.$queryRaw`
      SELECT 
        SUM(order_count) as total_orders,
        SUM(total_revenue) as total_revenue,
        AVG(avg_order_value) as avg_order_value,
        SUM(delivered_count) as delivered_orders,
        SUM(delivered_revenue) as delivered_revenue
      FROM SalesMetrics 
      WHERE organizationId = ${organizationId}
        AND date BETWEEN ${startDate} AND ${endDate}
    `;
    
    // Query com índices otimizados
    const topProducts = await prisma.$queryRaw`
      SELECT 
        product_name,
        total_revenue,
        total_margin,
        avg_margin_percentage
      FROM CostAnalysis 
      WHERE organizationId = ${organizationId}
        AND month BETWEEN date_trunc('month', ${startDate}) 
        AND date_trunc('month', ${endDate})
      ORDER BY total_revenue DESC
      LIMIT 10
    `;
    
    return { salesMetrics, topProducts };
  }
  
  // Query paginada para grandes volumes
  static async getPaginatedReport(
    prisma: PrismaClient,
    organizationId: string,
    filters: any,
    page: number = 1,
    limit: number = 100
  ) {
    const offset = (page - 1) * limit;
    
    const [data, total] = await Promise.all([
      prisma.$queryRaw`
        SELECT * FROM CostAnalysis 
        WHERE organizationId = ${organizationId}
        ORDER BY total_revenue DESC
        LIMIT ${limit} OFFSET ${offset}
      `,
      prisma.$queryRaw`
        SELECT COUNT(*) as count FROM CostAnalysis 
        WHERE organizationId = ${organizationId}
      `
    ]);
    
    return {
      data,
      pagination: {
        page,
        limit,
        total: Number(total[0].count),
        pages: Math.ceil(Number(total[0].count) / limit)
      }
    };
  }
}
```

```typescript
// JobQueue.ts
export class JobQueue {
  private queue: Queue;
  
  constructor() {
    this.queue = new Queue('analytics-jobs', {
      redis: { host: 'localhost', port: 6379 },
      defaultJobOptions: {
        removeOnComplete: 10,
        removeOnFail: 5,
        attempts: 3,
        backoff: 'exponential'
      }
    });
    
    this.setupProcessors();
  }
  
  private setupProcessors() {
    // Processamento de relatórios pesados
    this.queue.process('heavy-report', 5, async (job) => {
      const { reportId, type, organizationId, filters } = job.data;
      
      job.progress(10);
      const data = await this.generateReportData(type, organizationId, filters);
      
      job.progress(50);
      const formatted = await this.formatReportData(data, type);
      
      job.progress(80);
      await this.saveReport(reportId, formatted);
      
      job.progress(100);
      await this.notifyReportReady(reportId);
    });
    
    // Refresh de views materializadas
    this.queue.process('refresh-views', async (job) => {
      const { organizationId } = job.data;
      await this.refreshMaterializedViews(organizationId);
    });
  }
  
  async scheduleHeavyReport(
    reportId: string,
    type: ReportType,
    organizationId: string,
    filters: any
  ): Promise<void> {
    await this.queue.add('heavy-report', {
      reportId,
      type,
      organizationId,
      filters
    }, {
      priority: this.getReportPriority(type),
      delay: 0
    });
  }
}
```

```typescript
// PerformanceMonitor.ts
export class PerformanceMonitor {
  
  static async trackQueryPerformance<T>(
    queryName: string,
    queryFn: () => Promise<T>
  ): Promise<T> {
    const startTime = Date.now();
    
    try {
      const result = await queryFn();
      const duration = Date.now() - startTime;
      
      // Log performance metrics
      console.log(`Query ${queryName} completed in ${duration}ms`);
      
      // Alert se muito lento
      if (duration > 5000) {
        await this.sendSlowQueryAlert(queryName, duration);
      }
      
      return result;
    } catch (error) {
      const duration = Date.now() - startTime;
      console.error(`Query ${queryName} failed after ${duration}ms:`, error);
      throw error;
    }
  }
  
  private static async sendSlowQueryAlert(queryName: string, duration: number) {
    // Implementar alerta para queries lentas
    console.warn(`SLOW QUERY ALERT: ${queryName} took ${duration}ms`);
  }
}
```

```typescript
// ReportAccessControl.ts
export class ReportAccessControl {
  
  static canAccessReport(user: User, reportType: ReportType): boolean {
    const permissions = {
      'DASHBOARD': ['OWNER', 'ADMIN', 'MANAGER'],
      'COST_ANALYSIS': ['OWNER', 'ADMIN', 'MANAGER'],
      'MATERIAL_ANALYTICS': ['OWNER', 'ADMIN', 'MANAGER', 'OPERATOR'],
      'DEMAND_FORECAST': ['OWNER', 'ADMIN', 'MANAGER']
    };
    
    return permissions[reportType]?.includes(user.role) || false;
  }
  
  static async auditReportAccess(
    userId: string,
    reportType: ReportType,
    filters: any,
    action: 'VIEW' | 'EXPORT'
  ): Promise<void> {
    await prisma.auditLog.create({
      data: {
        userId,
        action: `REPORT_${action}`,
        tableName: 'Reports',
        recordId: reportType,
        newValues: { reportType, filters },
        ipAddress: '0.0.0.0', // Obter do request
        userAgent: 'Analytics System'
      }
    });
  }
}
```
